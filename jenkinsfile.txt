pipeline {
    agent any

    stages {
        stage('Checkout') {
            steps {
                script {
                    git url: 'https://github.com/akshitttt/Jenkins_Spring.git'
                    sh "ls -lart ./*"
                    sh "git branch -a"
                }
            }
       }

        stage ('Build Stage') {
            withMaven(maven: 'maven') {
                sh 'mvn clean compile'
            }
        }

        stage ('Testing Stage') {
            withMaven(maven: 'maven') {
                sh 'mvn test'
            }
        }
    }
}
